/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode mergeKLists(ListNode[] lists) {
        int n = lists.length;
        ListNode head = null;
        ListNode result = null;
        
        while(true) {
            int index = 0;
            boolean isNull = true;
            Integer lowest = null;
            for(int i = 0; i < n; i++) {
                ListNode node = lists[i];

                if(node == null) {
                    continue;
                }

                isNull = false;

                if(head != null && head.val == node.val) {
                    index = i;
                    lowest = node.val;
                    break;
                }
                

                if(lowest == null) {
                    lowest = node.val;
                    index = i;
                }

                if(lowest > node.val) {
                    index = i;
                    lowest = node.val;
                }
            }

            if(isNull) {
                break;
            }

            ListNode node = lists[index];
            node = node.next;
            lists[index] = node;
            
            if(head == null) {
                head = new ListNode(lowest);
                result = head;
            } else {
                head.next = new ListNode(lowest);
                head = head.next;
            }
        }

        return result;
    }
}
